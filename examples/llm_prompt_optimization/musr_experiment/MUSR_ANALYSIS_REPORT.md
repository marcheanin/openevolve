def evaluate_musr_response(model_answer: str, correct_answer: str) -> bool:
    """
    Multiple choice: Ğ¾Ñ‚Ğ²ĞµÑ‚ A, B, C, D Ğ¸Ğ»Ğ¸ E

def calculate_accuracy_by_task(results: list) -> dict:
    """Ğ’Ñ‹Ñ‡Ğ¸ÑĞ»ÑĞµĞ¼ accuracy Ğ¿Ğ¾ Ñ‚Ğ¸Ğ¿Ğ°Ğ¼ Ğ·Ğ°Ğ´Ğ°Ñ‡"""
    task_results = {

        'team_allocation': {'correct': 0, 'total': 0}
